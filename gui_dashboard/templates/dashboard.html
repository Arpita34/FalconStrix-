      Process Creation & Management

      fork(), wait(), Parent-Child Architecture

      How It Works

      The main.cpp C++ engine uses fork() to create a child process. The parent acts as a supervisor while the child runs 3 monitoring threads. waitpid() collects exit status to prevent zombie processes and enable crash-restart.

      Source Code — main.cpp (complete)

      #include <iostream>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <pthread.h>
#include <signal.h>
#include <cstdlib>
#include "process_monitor.h"
#include "behavior_detector.h"
#include "resource_monitor.h"

// 5. Signal Handling — async-signal-safe flag
volatile sig_atomic_t running = 1;
static pid_t child_pid = -1;

// 5. Parent signal handler
void signal_handler(int signum) {
    if (signum == SIGINT) {
        std::cout << "\n[SIGNAL] Caught SIGINT (Ctrl+C), initiating graceful shutdown..." << std::endl;
    } else if (signum == SIGTERM) {
        std::cout << "\n[SIGNAL] Caught SIGTERM, initiating graceful shutdown..." << std::endl;
    }
    running = 0;
    // Forward signal to child for coordinated shutdown
    if (child_pid > 0) {
        kill(child_pid, signum);
    }
}

// 5. Child signal handler
void child_signal_handler(int signum) {
    std::cout << "[SIGNAL] Child received signal " << signum << ", stopping threads..." << std::endl;
    running = 0;
}

// 2. Multithreading — child engine runs 3 POSIX threads
void run_child_engine() {
    signal(SIGINT, child_signal_handler);
    signal(SIGTERM, child_signal_handler);

    pthread_t monitor_thread, detector_thread, resource_thread;

    // Thread 1: Process Monitor (enumerates /proc)
    pthread_create(&monitor_thread, NULL, start_process_monitor, NULL);
    // Thread 2: Behavior Detector (anomaly scanning)
    pthread_create(&detector_thread, NULL, start_behavior_detector, NULL);
    // Thread 3: Resource Monitor (CPU & Memory)
    pthread_create(&resource_thread, NULL, start_resource_monitor, NULL);

    pthread_join(monitor_thread, NULL);
    pthread_join(detector_thread, NULL);
    pthread_join(resource_thread, NULL);

    std::cout << "[*] Child engine: All worker threads stopped." << std::endl;
}

// 1. Process Creation — fork() + 10. Fault Tolerance — restart loop
int main() {
    std::cout << "[*] Starting Blue Team OS Engine (Parent PID: " << getpid() << ")" << std::endl;
    signal(SIGINT, signal_handler);
    signal(SIGTERM, signal_handler);

    while (running) {
        pid_t pid = fork();  // 1. Process Creation
        if (pid < 0) {
            std::cerr << "[!] fork() failed." << std::endl;
            return 1;
        }
        if (pid == 0) {
            // CHILD PROCESS — runs monitoring engine
            std::cout << "[*] Child engine started (PID: " << getpid() << ")" << std::endl;
            run_child_engine();
            _exit(0);
        } else {
            // PARENT PROCESS — supervises child
            child_pid = pid;
            int status;
            waitpid(pid, &status, 0);  // 1. Zombie prevention

            if (!running) break;  // Graceful shutdown

            // 10. Fault Tolerance — restart crashed child
            if (WIFEXITED(status) && WEXITSTATUS(status) != 0) {
                std::cerr << "[!] Child exited with error (code "
                          << WEXITSTATUS(status) << "), restarting in 2s..." << std::endl;
                sleep(2);
                continue;
            } else if (WIFSIGNALED(status)) {
                std::cerr << "[!] Child killed by signal " << WTERMSIG(status)
                          << ", restarting in 2s..." << std::endl;
                sleep(2);
                continue;
            }
            break;  // Normal exit
        }
    }
    std::cout << "[*] Parent shutting down." << std::endl;
    return 0;
}

      Multithreading (Concurrency)

      POSIX Threads — pthread_create, pthread_join

      How It Works

      Three threads run concurrently inside the child process: Process Monitor, Behavior Detector, and Resource Monitor. Each runs an infinite loop performing its task with periodic sleep() intervals.

      Thread Entry Points (from source files)

      // ─── process_monitor.cpp ─── Thread 1
void* start_process_monitor(void* arg) {
    std::cout << "[+] Process Monitor thread started (TID in PID: "
              << getpid() << ")" << std::endl;
    while(true) {
        int count = get_process_count();
        std::cout << "[MONITOR] Process count: " << count << std::endl;
        if (count > 500) {
            std::cout << "[MONITOR] Warning: High process count ("
                      << count << ")" << std::endl;
        }
        sleep(5);
    }
    return NULL;
}

// ─── behavior_detector.cpp ─── Thread 2
void* start_behavior_detector(void* arg) {
    std::cout << "[+] Behavior Detector thread started." << std::endl;
    int last_process_cnt = get_process_count();

    while(true) {
        int cur_cnt = get_process_count();

        // Detect rapid process spike (>20 new in one cycle)
        if ((cur_cnt - last_process_cnt) > 20) {
            send_event_to_backend("PROCESS_SPIKE",
                "Detected rapid process count spike",
                getpid(), "system", 3);
        }
        last_process_cnt = cur_cnt;

        // Scan for reverse shell processes (nc, ncat, netcat)
        auto pids = get_all_pids_safe();  // mutex-protected
        for (int p : pids) {
            std::string pname = get_process_name(p);
            if (pname == "nc" || pname == "ncat" || pname == "netcat") {
                send_event_to_backend("SUSPICIOUS_PROC",
                    "Detected reverse shell: " + pname, p, pname, 3);
            }
        }
        sleep(2);
    }
    return NULL;
}

// ─── resource_monitor.cpp ─── Thread 3
void* start_resource_monitor(void* arg) {
    std::cout << "[+] Resource Monitor thread started." << std::endl;
    get_cpu_usage();  // prime the delta
    sleep(1);

    while (true) {
        CpuUsage cpu = get_cpu_usage();
        MemoryInfo mem = get_memory_info();

        std::cout << "[RES] CPU: " << cpu.usage_percent << "% | MEM: "
                  << mem.usage_percent << "%" << std::endl;

        if (cpu.usage_percent > 90.0) {
            send_event_to_backend("HIGH_CPU",
                "CPU usage exceeded 90%", getpid(), "resource_monitor", 3);
        }
        if (mem.usage_percent > 90.0) {
            send_event_to_backend("HIGH_MEMORY",
                "Memory usage exceeded 90%", getpid(), "resource_monitor", 3);
        }
        sleep(5);
    }
    return NULL;
}

      Synchronization (Mutex)

      Critical Section Protection, Race Condition Prevention

      How It Works

      A pthread_mutex_t guards access to /proc directory enumeration. Both process monitor and behavior detector threads acquire the lock before reading process data, preventing data races on the shared PID list.

      Source Code — process_monitor.cpp

      // Global mutex — shared between all threads
pthread_mutex_t proc_mut = PTHREAD_MUTEX_INITIALIZER;

// Raw /proc enumeration (NOT thread-safe)
std::vector<int> get_all_pids() {
    std::vector<int> pids;
    DIR* dir = opendir("/proc");
    if (!dir) return pids;

    struct dirent* entry;
    while ((entry = readdir(dir)) != NULL) {
        if (entry->d_type == DT_DIR) {
            std::string name = entry->d_name;
            bool is_pid = true;
            for (char c : name) {
                if (!isdigit(c)) { is_pid = false; break; }
            }
            if (is_pid && !name.empty()) {
                pids.push_back(std::stoi(name));
            }
        }
    }
    closedir(dir);
    return pids;
}

// Mutex-protected process count
int get_process_count() {
    pthread_mutex_lock(&proc_mut);
    int count = get_all_pids().size();
    pthread_mutex_unlock(&proc_mut);
    return count;
}

// Mutex-protected PID list access (thread-safe)
std::vector<int> get_all_pids_safe() {
    pthread_mutex_lock(&proc_mut);
    std::vector<int> pids = get_all_pids();
    pthread_mutex_unlock(&proc_mut);
    return pids;
}

      Signal Handling

      SIGINT, SIGTERM, Graceful Shutdown

      How It Works

      Both parent and child register signal handlers. On Ctrl+C (SIGINT) or kill command (SIGTERM), the handler sets running = 0 and forwards the signal to the child for coordinated shutdown. Uses volatile sig_atomic_t for async-signal-safety.

      Source Code — main.cpp

// Parent signal handler — SIGINT (Ctrl+C) and SIGTERM
void signal_handler(int signum) {
    if (signum == SIGINT) {
        std::cout << "\n[SIGNAL] Caught SIGINT (Ctrl+C), "
                  << "initiating graceful shutdown..." << std::endl;

      Resource Monitoring

      CPU & Memory Tracking via /proc

      How It Works

      A dedicated thread reads /proc/stat for CPU usage (delta between samples) and /proc/meminfo for memory stats. Per-process data via /proc/[pid]/stat. Alerts when usage exceeds 90%.

      Source Code — resource_monitor.cpp (complete)

      #include "resource_monitor.h"
#include <fstream>
#include <sstream>
#include <iostream>
#include <unistd.h>

static unsigned long long prev_total = 0;
static unsigned long long prev_idle = 0;

// 8. Read CPU usage from /proc/stat
CpuUsage get_cpu_usage() {
    CpuUsage cpu = {0, 0, 0, 0, 0};
    std::ifstream stat_file("/proc/stat");
    if (!stat_file.is_open()) return cpu;

    std::string line;
    std::getline(stat_file, line);
    std::istringstream iss(line);
    std::string cpu_label;
    unsigned long long user, nice, system, idle, iowait, irq, softirq, steal;
    iss >> cpu_label >> user >> nice >> system >> idle
        >> iowait >> irq >> softirq >> steal;

    cpu.user = user; cpu.nice = nice;
    cpu.system = system; cpu.idle = idle;

    unsigned long long total_idle = idle + iowait;
    unsigned long long total = user + nice + system + idle
        + iowait + irq + softirq + steal;

    unsigned long long diff_idle = total_idle - prev_idle;
    unsigned long long diff_total = total - prev_total;
    if (diff_total > 0) {
        cpu.usage_percent = (double)(diff_total - diff_idle)
            / (double)diff_total * 100.0;
    }
    prev_idle = total_idle;
    prev_total = total;
    return cpu;
}

// 8. Read memory info from /proc/meminfo
MemoryInfo get_memory_info() {
    MemoryInfo mem = {0, 0, 0, 0, 0, 0.0};
    std::ifstream meminfo("/proc/meminfo");
    if (!meminfo.is_open()) return mem;

    std::string line;
    while (std::getline(meminfo, line)) {
        std::istringstream iss(line);
        std::string key;
        unsigned long value;
        iss >> key >> value;
        if (key == "MemTotal:")          mem.total_kb = value;
        else if (key == "MemFree:")      mem.free_kb = value;
        else if (key == "MemAvailable:") mem.available_kb = value;
        else if (key == "Buffers:")      mem.buffers_kb = value;
        else if (key == "Cached:")       mem.cached_kb = value;
    }
    if (mem.total_kb > 0) {
        unsigned long used = mem.total_kb - mem.available_kb;
        mem.usage_percent = (double)used / (double)mem.total_kb * 100.0;
    }
    return mem;
}

// 8. Resource monitor thread — runs every 5 seconds
void* start_resource_monitor(void* arg) {
    std::cout << "[+] Resource Monitor thread started." << std::endl;
    get_cpu_usage();  // Prime the delta calculation
    sleep(1);

    while (true) {
        CpuUsage cpu = get_cpu_usage();
        MemoryInfo mem = get_memory_info();

        std::cout << "[RES] CPU: " << cpu.usage_percent
                  << "% | MEM: " << mem.usage_percent << "%"
                  << " (" << (mem.total_kb - mem.available_kb) / 1024
                  << "MB / " << mem.total_kb / 1024 << "MB)"
                  << std::endl;

        if (cpu.usage_percent > 90.0) {
            send_event_to_backend("HIGH_CPU",
                "CPU usage exceeded 90% threshold",
                getpid(), "resource_monitor", 3);
        }
        if (mem.usage_percent > 90.0) {
            send_event_to_backend("HIGH_MEMORY",
                "Memory usage exceeded 90% threshold",
                getpid(), "resource_monitor", 3);
        }
        sleep(5);
    }
    return NULL;
}

      def active_defense():
    """Invoked when FSM State -> LOCKED.
    Kills all processes linked to unresolved HIGH/CRITICAL alerts."""
    query = """
    SELECT DISTINCT p.pid, p.process_name, e.event_id
    FROM Processes p
    JOIN Events e ON p.process_id = e.process_id
    JOIN Alerts a ON e.event_id = a.event_id
    JOIN Severity s ON a.severity_id = s.severity_id
    WHERE a.is_resolved = FALSE
      AND s.level_name IN ('HIGH', 'CRITICAL', 'MEDIUM')
    """
    bad_procs = fetch_query(query)

    for proc in bad_procs:
        pid = proc['pid']
        name = proc['process_name']
        try:
            os.kill(pid, signal.SIGKILL)
            msg = f"Terminated malicious process {name} (PID: {pid})"
            print(f"[*] RESPONSE ENGINE: {msg}")
            execute_query(
                "INSERT INTO Events (event_type, description, source) "
                "VALUES (%s, %s, %s)",
                ('RESPONSE_ACTION', msg, 'Response_Engine'))
            execute_query(
                "UPDATE Alerts SET is_resolved=TRUE WHERE event_id=%s",
                (proc['event_id'],))
        except ProcessLookupError:
            pass
        except PermissionError:
            print(f"[!] Permission denied to kill PID {pid}")
